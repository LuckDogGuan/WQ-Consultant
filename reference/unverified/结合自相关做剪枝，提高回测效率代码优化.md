首先对大佬分享的代码进行修改，定义了cal_self_correlation方法，将相关系矩阵输出保存，这个方法中涉及到的调用其他方法可以去链接帖子看。

```python
def cal_self_correlation(alpha_ids_to_check,years_of_data) -> pd.DataFrame:    sess = login()    if not sess:        logging.error("无法建立会话，程序退出。")        exit()    print(f"\n--- 步骤 1: 开始获取 {len(alpha_ids_to_check)} 个 Alpha 的 PnL 数据 ---")    pnl_dataframes = []    failed_alphas = []    i = 1    for alpha_id in alpha_ids_to_check:        try:            pnl_df = _get_alpha_pnl(sess,alpha_id)            if not pnl_df.empty:                pnl_dataframes.append(pnl_df)                print(f"成功获取第{i}个Alpha '{alpha_id}' PnL 数据。")                i = i + 1            else:                print(f"获取的 Alpha '{alpha_id}' PnL 数据为空，将跳过。")                failed_alphas.append(alpha_id)                i = i + 1        except Exception as e:            print(f"获取 Alpha '{alpha_id}' 数据时失败: {e}")            failed_alphas.append(alpha_id)            i= i+1    if len(pnl_dataframes) < 2:        print("成功获取的 Alpha 数量不足2个，无法进行相关性计算。程序退出。")        exit()    print(f"--- 成功获取 {len(pnl_dataframes)} 个 Alpha 的数据。失败: {failed_alphas or '无'} ---")    print("\n--- 步骤 2: 合并所有 PnL 数据并计算相关性矩阵 ---")    # 使用 pd.concat 高效合并所有DataFrame，'inner' join确保只保留所有alpha共有的日期    # 这是比循环 merge 更高效的方法    combined_pnl = pd.concat(pnl_dataframes, axis=1, join='inner')    if combined_pnl.empty or len(combined_pnl) < 2:        print("所有 Alpha 没有共同的交易日期，或共同日期少于2天，无法计算相关性。")        exit()    # 筛选指定时间窗口内的数据    cutoff_date = combined_pnl.index.max() - pd.DateOffset(years=years_of_data)    combined_pnl = combined_pnl[combined_pnl.index > cutoff_date]    # 计算所有 alpha 的日收益率 (rets = pnl_t - pnl_{t-1})    # 这种向量化操作远快于逐个计算    combined_rets = combined_pnl - combined_pnl.ffill().shift(1)    # 删除因 shift(1) 操作产生的 NaN 值    combined_rets.dropna(inplace=True)    if len(combined_rets) < 2:        logging.error(f"在 {years_of_data} 年时间窗口内，有效数据点不足2个，无法计算。")        exit()    # 一次性计算出所有 alpha 两两之间的相关性矩阵    correlation_matrix = combined_rets.corr()    # =================================================================    # 步骤三: 打印结果    # =================================================================    print("\n\n" + "=" * 25 + " Alpha 相关性矩阵 " + "=" * 25)    print(f"基于过去 {years_of_data} 年的日收益率计算得出：")    print("高相关 (>=0.7) 标记为红色, 中等相关 (>=0.5) 标记为蓝色。")    print("-" * 75)    # *** 调用新的打印函数，替换掉原来的 print(correlation_matrix) ***    print_colored_matrix(correlation_matrix)    return correlation_matrix    print("-" * 75)
```

定义剪枝方法，我在其中使用了margin和fitness作为筛选的条件，这个可以结合自己的想法修改。

```python
def prune_alphas(correlation_matrix, alpha_list):    """    筛选出相关性低于0.7且margin和fitness综合得分较高的alpha列表    参数:        correlation_matrix: pd.DataFrame，相关性矩阵，行和列均为alpha_id，值为相关系数        alpha_list: 双层嵌套列表，每个元素为alpha信息列表，格式为                   [alpha_id, exp, sharpe, turnover, fitness, margin, dateCreated, decay]    返回:        list: 筛选后的alpha信息列表    """    # 创建alpha_id到其信息的映射，方便快速查找    alpha_dict = {alpha[0]: alpha for alpha in alpha_list}    # 提取所有alpha_id（确保与相关性矩阵一致）    valid_alpha_ids = [alpha[0] for alpha in alpha_list]    # 计算每个alpha的综合得分 (margin + fitness) * 0.5    alpha_scores = {        alpha_id: 0.5 * alpha[4] + 0.5 * alpha[5]  # alpha[4]是fitness，alpha[5]是margin        for alpha_id, alpha in alpha_dict.items()    }    # 按综合得分降序排序alpha，优先考虑高分alpha    sorted_alpha_ids = sorted(alpha_scores.keys(), key=lambda x: alpha_scores[x], reverse=True)    # 用于存储最终选中的alpha    selected = []    for alpha_id in sorted_alpha_ids:        # 检查当前alpha与已选中alpha的相关性（排除自身）        conflict = False        for selected_id in selected:            # 跳过与自身的相关性检查            if alpha_id == selected_id:                continue            # 获取相关系数并转换为浮点数            corr = float(correlation_matrix.loc[alpha_id, selected_id])            # 如果相关性高于等于0.7，视为冲突            if corr >= 0.8:                conflict = True                break        # 如果没有冲突，加入选中列表        if not conflict:            selected.append(alpha_id)    # 从原始列表中提取选中的alpha信息，保持原始顺序    result = [alpha for alpha in alpha_list if alpha[0] in selected]    return result
```

调用剪枝方法

```python
prune_after_alphas = prune_alphas(correlation_matrix,fo_tracker)
```
